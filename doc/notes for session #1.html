#1 setup project

rvm use --create 1.9.3@codemasters

gem install rails
rails new codemasters -T (to skip test::unit since we're using rspec)

cd codemasters
rvm --rvmrc 1.9.3@codemasters
cd ..
cd -

git init .
git add .
git ci

#2 gitignore IDEA and OSX files

#3 configure database

add mysql2 to gemfile
bundle install

be rake db:create

#4 setup devise

add devise do gemfile
bundle install

rails g devise:install

follow instructions

rails g controller home index --skip-stylesheets

rm public/index.html

rails g devise user name:string location:string master:boolean

be rake db:migrate

rails g devise:views

add some validations to user model, i18n and a logout button

#5 cucumber my way out of master registration and listing the user profile

add to Gemfile:

group :development, :test do
  gem 'rspec-rails'
  gem 'cucumber-rails'
  gem 'database_cleaner'
  gem 'launchy'
end

bundle install
rails g cucumber:install
